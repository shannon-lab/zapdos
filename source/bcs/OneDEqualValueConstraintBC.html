OneDEqualValueConstraintBC

Computes the integral of lambda times dg term from the mortar method (for two 1D domains only).

The constraint is imposed using a Lagrange multiplier, a scalar variable.

More information about the mortar method may be found in the Constraints documentation.

Example input syntax

In this example, two 1D lines are connected using the OneDEqualValueConstraintBC as a mortar constraint. The sign to impose on the variable value on the boundary nodes is specified with the "vg" parameter. The Lagrange multiplier, lm, is also subjected to a NodalEqualValueConstraint to achieve the continuity at the interface.

[BCs<<<{"href": "../../syntax/BCs/index.html"}>>>]
  [./left]
    type = DirichletBC<<<{"description": "Imposes the essential boundary condition $u=g$, where $g$ is a constant, controllable value.", "href": "DirichletBC.html"}>>>
    variable<<<{"description": "The name of the variable that this residual object operates on"}>>> = u
    boundary<<<{"description": "The list of boundary IDs from the mesh where this object applies"}>>> = '1'
    value<<<{"description": "Value of the BC"}>>> = 1
  [../]

  [./right]
    type = DirichletBC<<<{"description": "Imposes the essential boundary condition $u=g$, where $g$ is a constant, controllable value.", "href": "DirichletBC.html"}>>>
    variable<<<{"description": "The name of the variable that this residual object operates on"}>>> = u
    boundary<<<{"description": "The list of boundary IDs from the mesh where this object applies"}>>> = '2'
    value<<<{"description": "Value of the BC"}>>> = 3
  [../]

  [./evc1]
    type = OneDEqualValueConstraintBC<<<{"description": "Computes the integral of lambda times dg term from the mortar method (for two 1D domains only).", "href": "OneDEqualValueConstraintBC.html"}>>>
    variable<<<{"description": "The name of the variable that this residual object operates on"}>>> = u
    boundary<<<{"description": "The list of boundary IDs from the mesh where this object applies"}>>> = '100'
    lambda<<<{"description": "Lagrange multiplier"}>>> = lm
    component<<<{"description": "Component of the Lagrange multiplier"}>>> = 0
    vg<<<{"description": "Variation of the constraint g wrt this surface (+1 or -1). Note: g = value1 - value2 = 0 "}>>> = 1
  [../]

  [./evc2]
    type = OneDEqualValueConstraintBC<<<{"description": "Computes the integral of lambda times dg term from the mortar method (for two 1D domains only).", "href": "OneDEqualValueConstraintBC.html"}>>>
    variable<<<{"description": "The name of the variable that this residual object operates on"}>>> = u
    boundary<<<{"description": "The list of boundary IDs from the mesh where this object applies"}>>> = '101'
    lambda<<<{"description": "Lagrange multiplier"}>>> = lm
    component<<<{"description": "Component of the Lagrange multiplier"}>>> = 0
    vg<<<{"description": "Variation of the constraint g wrt this surface (+1 or -1). Note: g = value1 - value2 = 0 "}>>> = -1
  [../]
[]
(moose/test/tests/mortar/1d/1d.i)

[Mesh]
  file = 2-lines.e
  construct_side_list_from_node_list = true
[]

[Variables]
  [./u]
    order = FIRST
    family = LAGRANGE
    block = '1 2'
  [../]

  [./lm]
    order = FIRST
    family = SCALAR
  [../]
[]

[Kernels]
  [./diff]
    type = Diffusion
    variable = u
  [../]
[]

[Problem]
  extra_tag_vectors = 'ref'
[]

[ScalarKernels]
  [./ced]
    type = NodalEqualValueConstraint
    variable = lm
    var = u
    boundary = '100 101'
    absolute_value_vector_tags = 'ref'
  [../]
[]

[BCs]
  [./left]
    type = DirichletBC
    variable = u
    boundary = '1'
    value = 1
  [../]

  [./right]
    type = DirichletBC
    variable = u
    boundary = '2'
    value = 3
  [../]

  [./evc1]
    type = OneDEqualValueConstraintBC
    variable = u
    boundary = '100'
    lambda = lm
    component = 0
    vg = 1
  [../]

  [./evc2]
    type = OneDEqualValueConstraintBC
    variable = u
    boundary = '101'
    lambda = lm
    component = 0
    vg = -1
  [../]
[]

[Preconditioning]
  [./fmp]
    type = SMP
    full = true
    solve_type = 'NEWTON'
  [../]
[]

[Executioner]
  type = Steady
[]

[Outputs]
  exodus = true
[]
