Example Input File Syntax


[Postprocessors]
  [electrode_ion_current]
    type = SideTotFluxIntegral
    variable = ions
    mobility = muions
    r = 0.0
    position_units = 1.0
    field_property_name = electric_field
    boundary = electrode
  []
[]
