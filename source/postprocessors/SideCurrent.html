Example Input File Syntax


[Postprocessors]
  [electrode_current]
    type = SideCurrent
    variable = electrons
    mean_en = electron_energy
    ions = ions
    field_property_name = electric_field
    r = 0.0
    position_units = 1.0
    mobility = electron_mu
    boundary = electrode
  []
[]
